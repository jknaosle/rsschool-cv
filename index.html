Code-example

Perimeter of squares in a rectangle from CODEWARS: The function perimeter has for parameter n where n + 1 is the number of squares (they are numbered from 0 to n) and returns the total perimeter of all the squares.

function perimeter(n) {
   let arr = [0,4]
   let sum = 4
   for (i=1; i <= n;i++){
      arr.push(arr[i]+arr[i-1])
      sum +=arr[i+1]
      }
   return sum
}
